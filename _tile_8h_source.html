52  void Tile<DEST, SRC>::operator()() noexcept {
53  Dimension dimension = Transformation<DEST, SRC>::destination->getDimension();
54  if (!dimension.isProper()) {
55  return; // nothing to do
56  }
57  // TAG: need offset support
58 
59  typename DestinationImage::Rows::RowIterator destRow = Transformation<DEST, SRC>::destination->getRows().getFirst();
60  typename SourceImage::ReadableRows::RowIterator srcBegin = Transformation<DEST, SRC>::source->getRows().getFirst();
61  typename SourceImage::ReadableRows::RowIterator srcEnd = Transformation<DEST, SRC>::source->getRows().getEnd();
62 
63  unsigned int verticalCycles = Transformation<DEST, SRC>::destination->getDimension().getHeight()/Transformation<DEST, SRC>::source->getDimension().getHeight();
64  unsigned int horizontalCycles = Transformation<DEST, SRC>::destination->getDimension().getWidth()/Transformation<DEST, SRC>::source->getDimension().getWidth();
65 
66  while (verticalCycles--) {
67  typename SourceImage::ReadableRows::RowIterator srcRow = srcBegin;
68  while (srcRow != srcEnd) {
69  typename DestinationImage::Rows::RowIterator::ElementIterator destColumn = destRow.getFirst();
70  typename SourceImage::ReadableRows::RowIterator::ElementIterator srcColumnEnd = srcRow.getEnd();
71 
72  for (unsigned int cycles = horizontalCycles; cycles > 0; --cycles) {
73 // destColumn = copy(destColumn, srcRow.getFirst(), srcRow.getEnd());
74  typename SourceImage::ReadableRows::RowIterator::ElementIterator srcColumn = srcRow.getFirst();
75  while (srcColumn != srcColumnEnd) {
76  *destColumn++ = *srcColumn++;
77  }
78  }
79 
80 // copy(destColumn, destRow.getEnd(), srcRow.getFirst());
81  typename DestinationImage::Rows::RowIterator::ElementIterator destColumnEnd = destRow.getEnd();
82  typename SourceImage::ReadableRows::RowIterator::ElementIterator srcColumn = srcRow.getFirst();
83  while (destColumn != destColumnEnd) {
84  *destColumn++ = *srcColumn++;
85  }
86  ++destRow;
87  ++srcRow;
88  }
89  }
90 
91  typename DestinationImage::Rows::RowIterator destEnd = Transformation<DEST, SRC>::destination->getRows().getEnd();
92  typename SourceImage::ReadableRows::RowIterator srcRow = srcBegin;
93  while (destRow < destEnd) {
94  typename DestinationImage::Rows::RowIterator::ElementIterator destColumn = destRow.getFirst();
95  typename SourceImage::ReadableRows::RowIterator::ElementIterator srcColumnEnd = srcRow.getEnd();
96 
97  for (unsigned int cycles = horizontalCycles; cycles > 0; --cycles) {
98 // destColumn = copy(destColumn, srcRow.getFirst(), srcRow.getEnd());
99  typename SourceImage::ReadableRows::RowIterator::ElementIterator srcColumn = srcRow.getFirst();
100  while (srcColumn != srcColumnEnd) {
101  *destColumn++ = *srcColumn++;
102  }
103  }
104 
105 // copy(destColumn, destRow.getEnd(), srcRow.getFirst());
106  typename DestinationImage::Rows::RowIterator::ElementIterator destColumnEnd = destRow.getEnd();
107  typename SourceImage::ReadableRows::RowIterator::ElementIterator srcColumn = srcRow.getFirst();
108  while (destColumn != destColumnEnd) {
109  *destColumn++ = *srcColumn++;
110  }
111  ++destRow;
112  ++srcRow;
113  }
114  }
