27  class HeatColorMap : public UnaryOperation<RGBPixel<long double>, long double> {
28  public:
29 
30  inline RGBPixel<long double> operator()(const long double& value) const noexcept {
31  RGBPixel<long double> result;
32  if (value <= 0) {
33  result.red = 0;
34  result.green = 0;
35  result.blue = 0;
36  return result;
37  }
38  long double temp = value * 3;
39  int section = static_cast<int>(temp);
40  long double fraction = temp - section;
41  switch (section) { // floor
42  case 0: // black -> dark orange
43  result.red = fraction;
44  result.green = fraction * 140.0/255;
45  result.blue = 0;
46  break;
47  case 1: // dark orange -> bright yellow
48  result.red = 1;
49  result.green = fraction * (1 - 140.0/255) + 140.0/255;
50  result.blue = fraction/2;
51  break;
52  case 2: // bright yellow -> white
53  result.red = 1;
54  result.green = 1;
55  result.blue = fraction/2 + 0.5;
56  break;
57  default: // clamp to white
58  result.red = 1;
59  result.green = 1;
60  result.blue = 1;
61  break;
62  }
63  return result;
64  }
65 
66  };
