69  void rotate(double alpha) noexcept {

110  const unsigned int width = Transformation<DestinationImage, SourceImage>::destination->getDimension().getWidth();
111  Pixel* dest = Transformation<DestinationImage, SourceImage>::destination->getElements();
112  Interpolate<Pixel> interpolate(*Transformation<DestinationImage, SourceImage>::source);
113 
114  // inverse of matrix (only x and y row)
115  double inverse[2][3];
116  const double factor = 1/(matrix[0][0] * matrix[1][1] - matrix[0][1] * matrix[1][0]);
117 
118  inverse[0][0] = matrix[1][1] * factor;
119  inverse[0][1] = -matrix[0][1] * factor;
120  inverse[0][2] = (matrix[0][1] * matrix[1][2] - matrix[0][2] * matrix[1][1]) * factor;
121  inverse[1][0] = -matrix[1][0] * factor;
122  inverse[1][1] = matrix[0][0] * factor;
123  inverse[1][2] = (matrix[0][2] * matrix[1][0] - matrix[0][0] * matrix[1][2]) * factor;
124 
125  for (unsigned int y = 0; y < height; ++y) {
126  double srcX = inverse[0][1] * y + inverse[0][2];
127  double srcY = inverse[1][1] * y + inverse[1][2];
128  for (unsigned int x = 0; x < width; ++x) {
129  srcX += inverse[0][0];
130  srcY += inverse[1][0];
131  *dest++ = static_cast<Pixel>(interpolate(srcX, srcY)); // TAG: round to nearest
132  }
133  }
