73  Statistic(const Image& image) noexcept
74  : numberOfSamples(image.getDimension().getSize()) {
75 
76  // count frequency of each pixel value
77  fill<unsigned int>(frequency, getArraySize(frequency), 0);
78 
79  typename Image::ReadableRows rows = Cast::implicit<const Image>(image).getRows();
80  typename Image::ReadableRows::RowIterator row = rows.getFirst();
81  for (; row != rows.getEnd(); ++row) {
82  typename Image::ReadableRows::RowIterator::ElementIterator column = row.getFirst();
83  for (; column != row.getEnd(); ++column) {
84  unsigned int value = *column;
85  if ((value >= PixelTraits<Pixel>::MINIMUM) && (value <= PixelTraits<Pixel>::MAXIMUM)) {
86  ++frequency[value];
87  }
88  }
89  }
90 
91  mean = 0;
92  for (unsigned int i = 0; i < NUMBER_OF_SYMBOLS; ++i) {
93  mean += Cast::implicit<double>(i) * frequency[i];
94  }
95  mean /= numberOfSamples;
96 
97  int count = numberOfSamples/2;
98  minimumFrequency = numberOfSamples;
99  maximumFrequency = 0;
100  minimum = PixelTraits<Pixel>::MAXIMUM;
101  maximum = PixelTraits<Pixel>::MINIMUM;
102  used = 0;
103  double sqrsum = 0;
104  entropy = numberOfSamples * Math::ln(Cast::implicit<double>(numberOfSamples));
105  for (unsigned int i = 0; i < NUMBER_OF_SYMBOLS; ++i) {
106  if (frequency[i] != 0) {
107  if (frequency[i] < minimumFrequency) {
108  minimumFrequency = frequency[i];
109  }
110  if (frequency[i] > maximumFrequency) {
111  maximumFrequency = frequency[i];
112  mode = i;
113  }
114  if (static_cast<int>(i) < minimum) {
115  minimum = i;
116  }
117  maximum = i;
118  if (count >= 0) {
119  count -= frequency[i];
120  median = i;
121  }
122  ++used;
123  sqrsum += frequency[i] * (i - mean) * (i - mean);
124  entropy -= frequency[i] * Math::ln(Cast::implicit<double>(frequency[i]));
125  } else {
126  minimumFrequency = 0;
127  }
128  }
129  if (numberOfSamples == 0) {
130  variance = 0;
131  } else {
132  variance = sqrsum/(numberOfSamples - 1);
133  entropy *= constant::LOG2E/numberOfSamples; // convert to binary units
134  }
135  }
