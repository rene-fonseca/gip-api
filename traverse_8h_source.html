57  inline void transform(LEFT& left, const RIGHT& right, BINOPR& function) {
58  // check if function is binary and write nice error if not???
59  bassert(
60  left.getDimension() == right.getDimension(),
61  ImageException("Images must have identical dimension")
62  );
63  typename LEFT::Rows rows = left.getRows();
64  typename LEFT::Rows::RowIterator row = rows.getFirst();
65  typename RIGHT::ReadableRows::RowIterator rightRow = right.getRows().getFirst();
66  while (row != rows.getEnd()) {
67  typename LEFT::Rows::RowIterator::ElementIterator column = row.getFirst();
68  typename RIGHT::ReadableRows::RowIterator::ElementIterator rightColumn = rightRow.getFirst();
69  while (column != row.getEnd()) {
70  *column = function(*column, *rightColumn);
71  ++column;
72  ++rightColumn;
73  }
74  ++row;
75  ++rightRow;
76  }
