104 template<class COMPONENT>
105 inline YCbCrPixel<COMPONENT> RGBToYCbCr(const RGBPixel<COMPONENT>& pixel) noexcept {
106  YCbCrPixel<COMPONENT> result;
107  typedef typename PixelTraits<RGBPixel<COMPONENT> >::Arithmetic Arithmetic;
108 
109  // see ITU-R recommendation BT (map to range [0; 1])
110  result.luma = 0.299 * static_cast<Arithmetic>(pixel.red) + 0.587 * static_cast<Arithmetic>(pixel.green) + 0.114 * static_cast<Arithmetic>(pixel.blue);
111 
112  result.cb = (pixel.blue - result.luma) * 0.5/(1.000-0.114); // map to range [-0.5; 0.5]
113  result.cr = (pixel.red - result.luma) * 0.5/(1.000-0.299); // map to range [-0.5; 0.5]
114  return result;
115 }

120 template<>
121 inline YCbCrPixel<unsigned char> RGBToYCbCr<unsigned char>(const RGBPixel<unsigned char>& pixel) noexcept {
122  YCbCrPixel<unsigned char> result;
123  typedef PixelTraits<RGBPixel<unsigned char> >::Arithmetic Arithmetic;
124  // we know that overflow isn't possible: (299 + 587 + 114) * 255 * 255 <= PrimitiveTraits<PixelTraits<ColorPixel>::Arithmetic>::MAXIMUM
125  // see ITU-R recommendation BT (map into range [0; 255])
126  Arithmetic temp = 299 * static_cast<Arithmetic>(pixel.red) + 587 * static_cast<Arithmetic>(pixel.green) + 114 * static_cast<Arithmetic>(pixel.blue);
127  result.luma = (temp + 1000/2)/1000; // round to nearest
128  result.cb = ((1000 * static_cast<Arithmetic>(pixel.blue) - temp) + (255+1)*(1000-114))/(2 * (1000-114)); // round to nearest
129  result.cr = ((1000 * static_cast<Arithmetic>(pixel.red) - temp) + (255+1)*(1000-299))/(2 * (1000-299)); // round to nearest
130  return result;
131 }

136 template<class COMPONENT>
137 inline RGBPixel<COMPONENT> YCbCrToRGB(const YCbCrPixel<COMPONENT>& pixel) noexcept {
138  typedef typename PixelTraits<YCbCrPixel<COMPONENT> >::Arithmetic Arithmetic;
139  RGBPixel<COMPONENT> result;
140  // map components into range [0; 1]
141  result.red = static_cast<Arithmetic>(pixel.luma) + 2 * (1.000-0.299) * static_cast<Arithmetic>(pixel.cr);
142  result.green = static_cast<Arithmetic>(pixel.luma) + (-0.114/0.587 * 2 * (1.000-0.114)) * static_cast<Arithmetic>(pixel.cb) + (-0.299/0.587 * 2 * (1.000-0.299)) * static_cast<Arithmetic>(pixel.cr);
143  result.blue = static_cast<Arithmetic>(pixel.luma) + 2 * (1.000-0.114) * static_cast<Arithmetic>(pixel.cb);
144  return result;
145 }

150 template<>
151 inline RGBPixel<unsigned char> YCbCrToRGB(const YCbCrPixel<unsigned char>& pixel) noexcept {
152  typedef PixelTraits<YCbCrPixel<unsigned char> >::Arithmetic Arithmetic;
153  RGBPixel<unsigned char> result;
154  Arithmetic temp;
155  temp = static_cast<Arithmetic>(pixel.luma) + (2*(1000-299) * static_cast<Arithmetic>(pixel.cr) - (1000-299)*255 + 1000/2)/1000; // round to nearest
156  result.red = (temp >= 0) ? ((temp <= 255) ? temp : 255) : 0;
157  temp = static_cast<Arithmetic>(pixel.luma) +
158  (
159  587 * 1000/2 + (114 * (1000-114)) * 255 + (299 * (1000-299)) * 255
160  - (114 * (1000-114)) * 2 * static_cast<Arithmetic>(pixel.cb)
161  - (299 * (1000-299)) * 2 * static_cast<Arithmetic>(pixel.cr)
162  )/(587*1000); // round to nearest
163  result.green = (temp >= 0) ? ((temp <= 255) ? temp : 255) : 0;
164  temp = static_cast<Arithmetic>(pixel.luma) + (2*(1000-114) * static_cast<Arithmetic>(pixel.cb) - (1000-114)*255 + 1000/2)/1000; // round to nearest
165  result.blue = (temp >= 0) ? ((temp <= 255) ? temp : 255) : 0;
166  return result;
167 }
