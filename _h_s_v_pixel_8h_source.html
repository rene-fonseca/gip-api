105 template<class COMPONENT>
106 inline HSVPixel<COMPONENT> RGBToHSV(const RGBPixel<COMPONENT>& pixel) noexcept {
107  typedef typename PixelTraits<RGBPixel<COMPONENT> >::Arithmetic Arithmetic;
108  HSVPixel<COMPONENT> result;
109 
110  COMPONENT max = maximum(pixel.red, pixel.green, pixel.blue);
111  COMPONENT min = minimum(pixel.red, pixel.green, pixel.blue);
112  typename PixelTraits<HSVPixel<COMPONENT> >::Arithmetic diff = max - min;
113 
114  result.value = max;
115  if (max == 0) { // TAG: problem if COMPONENT is floating-point type
116  result.saturation = 0;
117  } else {
118  result.saturation = diff/max;
119  }
120  if (result.saturation == 0) {
121  result.hue = 0; // undefined
122  return result;
123  }
124 
125  if (pixel.red == max) {
126  result.hue = (0 + (static_cast<Arithmetic>(pixel.green) - static_cast<Arithmetic>(pixel.blue))/diff)/6;
127  } else if (pixel.green == max) {
128  result.hue = (2 + (static_cast<Arithmetic>(pixel.blue) - static_cast<Arithmetic>(pixel.red))/diff)/6;
129  } else { // pixel.blue == max
130  result.hue = (4 + (static_cast<Arithmetic>(pixel.red) - static_cast<Arithmetic>(pixel.green))/diff)/6;
131  }
132  if (result.hue < 0) {
133  result.hue += 1;
134  }
135  return result;
136 }

141 template<>
142 inline HSVPixel<unsigned char> RGBToHSV<unsigned char>(const RGBPixel<unsigned char>& pixel) noexcept {
143  typedef PixelTraits<RGBPixel<unsigned char> >::Arithmetic Arithmetic;
144  HSVPixel<unsigned char> result;
145 
146  unsigned char max = maximum(pixel.red, pixel.green, pixel.blue);
147  unsigned char min = minimum(pixel.red, pixel.green, pixel.blue);
148  PixelTraits<HSVPixel<unsigned char> >::Arithmetic diff = max - min; // >= 0
149 
150  result.value = max;
151  if (max == 0) {
152  result.saturation = 0;
153  } else {
154  result.saturation = (2 * 255 * diff + max)/(2 * max); // round to nearest
155  }
156  if (result.saturation == 0) {
157  result.hue = 0; // undefined
158  return result;
159  }
160 
161  PixelTraits<HSVPixel<unsigned char> >::Arithmetic temp;
162  if (pixel.red == max) {
163  if (static_cast<Arithmetic>(pixel.green) - static_cast<Arithmetic>(pixel.blue) >= 0) {
164  temp = ((6/2 + 255 * 0) * diff + 255 * (static_cast<Arithmetic>(pixel.green) - static_cast<Arithmetic>(pixel.blue)))/(6 * diff);
165  } else {
166  temp = ((6/2 + 255 * 6) * diff + 255 * (static_cast<Arithmetic>(pixel.green) - static_cast<Arithmetic>(pixel.blue)))/(6 * diff);
167  }
168  } else if (pixel.green == max) {
169  temp = ((6/2 + 255 * 2) * diff + 255 * (static_cast<Arithmetic>(pixel.blue) - static_cast<Arithmetic>(pixel.red)))/(6 * diff);
170  } else { // pixel.blue == max
171  temp = ((6/2 + 255 * 4) * diff + 255 * (static_cast<Arithmetic>(pixel.red) - static_cast<Arithmetic>(pixel.green)))/(6 * diff);
172  }
173  ASSERT((temp >= 0) && (temp <= 255));
174  result.hue = temp;
175  return result;
176 }

181 template<class COMPONENT>
182 inline RGBPixel<COMPONENT> HSVToRGB(const HSVPixel<COMPONENT>& pixel) noexcept {
183  typedef typename PixelTraits<HSVPixel<COMPONENT> >::Arithmetic Arithmetic;
184  RGBPixel<COMPONENT> result;
185 
186  if (pixel.saturation == 0) { // TAG: problem if COMPONENT is floating-point type
187  result.red = pixel.value;
188  result.green = pixel.value;
189  result.blue = pixel.value;
190  return result;
191  }
192 
193  Arithmetic temp = static_cast<Arithmetic>(pixel.hue) * 360/60;
194  const int section = static_cast<int>(temp); // floor // TAG: problem if hue = 1 => section = 6 and fraction = 0
195  const Arithmetic fraction = temp - section;
196 
197  switch (section) {
198  case 0:
199  result.red = pixel.value;
200  result.green = pixel.value * (1.0 - (pixel.saturation * (1.0 - fraction)));
201  result.blue = pixel.value * (1.0 - pixel.saturation);
202  break;
203  case 1:
204  result.red = pixel.value * (1.0 - pixel.saturation * fraction);
205  result.green = pixel.value;
206  result.blue = pixel.value * (1.0 - pixel.saturation);
207  break;
208  case 2:
209  result.red = pixel.value * (1.0 - pixel.saturation);
210  result.green = pixel.value;
211  result.blue = pixel.value * (1.0 - (pixel.saturation * (1.0 - fraction)));
212  break;
213  case 3:
214  result.red = pixel.value * (1.0 - pixel.saturation);
215  result.green = pixel.value * (1.0 - pixel.saturation * fraction);
216  result.blue = pixel.value;
217  break;
218  case 4:
219  result.red = pixel.value * (1.0 - (pixel.saturation * (1.0 - fraction)));
220  result.green = pixel.value * (1.0 - pixel.saturation);
221  result.blue = pixel.value;
222  break;
223  default: // both 5 and 6
224  result.red = pixel.value;
225  result.green = pixel.value * (1.0 - pixel.saturation);
226  result.blue = pixel.value * (1.0 - pixel.saturation * fraction);
227  break;
228  }
229  return result;
230 }

235 template<>
236 inline RGBPixel<unsigned char> HSVToRGB(const HSVPixel<unsigned char>& pixel) noexcept {
237  typedef PixelTraits<HSVPixel<unsigned char> >::Arithmetic Arithmetic;
238  RGBPixel<unsigned char> result;
239 
240  if (pixel.saturation == 0) {
241  result.red = pixel.value;
242  result.green = pixel.value;
243  result.blue = pixel.value;
244  return result;
245  }
246 
247  Arithmetic temp = static_cast<Arithmetic>(pixel.hue) * 360/60;
248  const int section = temp/255; // floor // TAG: problem if hue = 255 => section = 6 and fraction = 0
249  const Arithmetic fraction = temp - section * 255; // scaled with 255
250 
251  switch (section) {
252  case 0:
253  result.red = pixel.value;
254  result.green = (255*255 + 2 * static_cast<Arithmetic>(pixel.value) * (255 * 255 - (pixel.saturation * (255 - fraction))))/(2 * 255 * 255);
255  result.blue = (255 + 2 * static_cast<Arithmetic>(pixel.value) * (255 - pixel.saturation))/(2 * 255);
256  break;
257  case 1:
258  result.red = (255*255 + 2 * static_cast<Arithmetic>(pixel.value) * (255 * 255 - pixel.saturation * fraction))/(2 * 255 * 255);
259  result.green = pixel.value;
260  result.blue = (255 + 2 * static_cast<Arithmetic>(pixel.value) * (255 - pixel.saturation))/(2 * 255);
261  break;
262  case 2:
263  result.red = (255 + 2 * static_cast<Arithmetic>(pixel.value) * (255 - pixel.saturation))/(2 * 255);
264  result.green = pixel.value;
265  result.blue = (255*255 + 2 * static_cast<Arithmetic>(pixel.value) * (255 * 255 - (pixel.saturation * (255 - fraction))))/(2 * 255 * 255);
266  break;
267  case 3:
268  result.red = (255 + 2 * static_cast<Arithmetic>(pixel.value) * (255 - pixel.saturation))/(2 * 255);
269  result.green = (255*255 + 2 * static_cast<Arithmetic>(pixel.value) * (255 * 255 - pixel.saturation * fraction))/(2 * 255 * 255);
270  result.blue = pixel.value;
271  break;
272  case 4:
273  result.red = (255*255 + 2 * static_cast<Arithmetic>(pixel.value) * (255 * 255 - (pixel.saturation * (255 - fraction))))/(2 * 255 * 255);
274  result.green = (255 + 2 * static_cast<Arithmetic>(pixel.value) * (255 - pixel.saturation))/(2 * 255);
275  result.blue = pixel.value;
276  break;
277  default: // both 5 and 6
278  result.red = pixel.value;
279  result.green = (255 + 2 * static_cast<Arithmetic>(pixel.value) * (255 - pixel.saturation))/(2 * 255);
280  result.blue = (255*255 + 2 * static_cast<Arithmetic>(pixel.value) * (255 * 255 - pixel.saturation * fraction))/(2 * 255 * 255);
281  break;
282  }
283  return result;
284 }
