55  void operator()() noexcept {
56  const unsigned int destWidth = Transformation<DEST, SRC>::destination->getDimension().getWidth();
57  const unsigned int destHeight = Transformation<DEST, SRC>::destination->getDimension().getHeight();
58  const unsigned int srcWidth = Transformation<DEST, SRC>::source->getDimension().getWidth();
59  const unsigned int srcHeight = Transformation<DEST, SRC>::source->getDimension().getHeight();
60  const double scale = 1.0/(srcHeight * srcWidth);
61 
62  typename DestinationImage::Rows::RowIterator destRow = Transformation<DEST, SRC>::destination->getRows().getFirst();
63  typename SourceImage::ReadableRows srcRowsLookup = Transformation<DEST, SRC>::source->getRows();
64  typename SourceImage::ReadableRows::RowIterator srcRow = srcRowsLookup.getFirst();
65  const typename SourceImage::ReadableRows::RowIterator srcEndRow = srcRowsLookup.getEnd();
66 
67  //Allocator<double> buffer(destWidth);
68  PrimitiveArray<double> buffer(destWidth);
69  fill<double>(buffer, destWidth, 0);
70 
71  unsigned int y = 0;
72  unsigned int rowModulus = destHeight;
73 
74  while (srcRow != srcEndRow) {
75  typename SourceImage::ReadableRows::RowIterator::ElementIterator srcColumn = srcRow.getFirst();
76  const typename SourceImage::ReadableRows::RowIterator::ElementIterator srcEndColumn = srcRow.getEnd();
77 
78  double* currentRow = buffer;
79  double integrate = 0;
80  unsigned int columnModulus = destWidth;
81 
82  if (rowModulus < srcHeight) {
83  while (srcColumn != srcEndColumn) {
84  if (columnModulus < srcWidth) {
85  integrate += destWidth * *srcColumn;
86  } else {
87  columnModulus -= srcWidth;
88  integrate += (destWidth - columnModulus) * *srcColumn;
89  *currentRow++ += destHeight * integrate;
90  integrate = columnModulus * *srcColumn;
91  }
92  columnModulus += destWidth;
93  ++srcColumn;
94  }
95  } else {
96  rowModulus -= srcHeight;
97  const unsigned int rowWeight = destHeight - rowModulus;
98  typename DestinationImage::Rows::RowIterator::ElementIterator destColumn = destRow.getFirst();
99  ++destRow;
100  while (srcColumn != srcEndColumn) {
101  if (columnModulus < srcWidth) {
102  integrate += destWidth * *srcColumn;
103  } else {
104  columnModulus -= srcWidth;
105  integrate += (destWidth - columnModulus) * *srcColumn;
106  *currentRow += rowWeight * integrate;
107  *destColumn++ = static_cast<typename DestinationImage::Pixel>(scale * *currentRow + 0.5); // TAG: round to nearest
108  *currentRow++ = rowModulus * integrate; // next row
109  integrate = columnModulus * *srcColumn;
110  }
111  columnModulus += destWidth;
112  ++srcColumn;
113  }
114  }
115  rowModulus += destHeight;
116  ++srcRow;
117  }
