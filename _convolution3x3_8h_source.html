348  typename SourceImage::ReadableRows::RowIterator endRow = rowLookup.getEnd();
349  typename SourceImage::ReadableRows::RowIterator previousRow = rowLookup.getFirst();
350  typename SourceImage::ReadableRows::RowIterator currentRow = previousRow + 1;
351  typename SourceImage::ReadableRows::RowIterator nextRow = currentRow + 1;
352 
353  typename DestinationImage::Rows::RowIterator destRow = Transformation<DEST, SRC>::destination->getRows().getFirst();
354 
355  // handle first row
356  ++destRow;
357 
358  while (nextRow < endRow) {
359  typename SourceImage::ReadableRows::RowIterator::ElementIterator previousRowColumn = previousRow.getFirst();
360  typename SourceImage::ReadableRows::RowIterator::ElementIterator currentRowColumn = currentRow.getFirst();
361  typename SourceImage::ReadableRows::RowIterator::ElementIterator nextRowColumn = nextRow.getFirst();
362  typename SourceImage::ReadableRows::RowIterator::ElementIterator endCurrentRowColumn = currentRow.getEnd() - 1;
363  typename DestinationImage::Rows::RowIterator::ElementIterator dest = destRow.getFirst();
364 
365  // first column
366  ++previousRowColumn;
367  ++currentRowColumn;
368  ++nextRowColumn;
369  ++dest;
370 
371  ApplyKernel<typename SourceImage::Pixel> applyKernel;
372  while (currentRowColumn < endCurrentRowColumn) {
373  *dest = applyKernel(previousRowColumn, currentRowColumn, nextRowColumn);
