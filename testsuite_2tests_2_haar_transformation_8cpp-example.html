class HaarToGray : public UnaryOperation<float, GrayPixel> {
public:
inline GrayPixel operator()(const float& value) const noexcept {
long double temp = value/2 + 128; // -255 <= value <= 255
if (temp < 0x00) {
return 0x00;
} else if (temp > 0xff) {
return 0xff;
} else {
return static_cast<GrayPixel>(temp);
}
}
};
