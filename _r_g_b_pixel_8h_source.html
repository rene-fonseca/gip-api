149  template<class COMPONENT>
150  inline RGBPixel<COMPONENT> blend(RGBPixel<COMPONENT> back, RGBPixel<COMPONENT> front, unsigned int opaque, unsigned int opacity) noexcept {
151  RGBPixel<COMPONENT> result;
152  unsigned int transparency = opaque - opacity;
153  result.red = (transparency * static_cast<typename PixelTraits<RGBPixel<COMPONENT> >::Arithmetic>(back.red) + opacity * static_cast<typename PixelTraits<RGBPixel<COMPONENT> >::Arithmetic>(front.red))/opaque;
154  result.green = (transparency * static_cast<typename PixelTraits<RGBPixel<COMPONENT> >::Arithmetic>(back.green) + opacity * static_cast<typename PixelTraits<RGBPixel<COMPONENT> >::Arithmetic>(front.green))/opaque;
155  result.blue = (transparency * static_cast<typename PixelTraits<RGBPixel<COMPONENT> >::Arithmetic>(back.blue) + opacity * static_cast<typename PixelTraits<RGBPixel<COMPONENT> >::Arithmetic>(front.blue))/opaque;
156  return result;
157  }

159  template<>
160  inline ColorPixel blend(ColorPixel back, ColorPixel front, unsigned int opaque, unsigned int opacity) noexcept {
161  ColorPixel result;
162  if (opaque == 2 * opacity) { // handle average
163  result.rgb = (((back.rgb ^ front.rgb) & 0xfefefefe) >> 1) + back.rgb & front.rgb; // no overflow
164  } else {
165  unsigned int transparency = opaque - opacity;
166  result.red = (2 * transparency * static_cast<PixelTraits<ColorPixel>::Arithmetic>(back.red) + 2 * opacity * static_cast<PixelTraits<ColorPixel>::Arithmetic>(front.red) + opaque)/(2 * opaque);
167  result.green = (2 * transparency * static_cast<PixelTraits<ColorPixel>::Arithmetic>(back.green) + 2 * opacity * static_cast<PixelTraits<ColorPixel>::Arithmetic>(front.green) + opaque)/(2 * opaque);
168  result.blue = (2 * transparency * static_cast<PixelTraits<ColorPixel>::Arithmetic>(back.blue) + 2 * opacity * static_cast<PixelTraits<ColorPixel>::Arithmetic>(front.blue) + opaque)/(2 * opaque);
169  }
170  return result;
171  }
